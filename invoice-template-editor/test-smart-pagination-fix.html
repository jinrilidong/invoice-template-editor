智能分页修复测试 - 内容溢出和重复问题

🐛 修复的问题

  • 问题1：当内容高度超过700px时，页面内容溢出700px外面
  • 问题2：第二页显示的内容是复制第一页的table section和Item section还有Description
  • 问题3：第二页应该显示第一页超过700px高度以外的内容，而不是重复内容

🔧 修复方案

1. 智能内容分配

// 计算每页应该显示的内容 const getPageContent = (pageIndex: number) => { const maxHeightPerPage = 700 let currentHeight = 0 const pageContent = { header: false, info: false, table: { show: false, rowsToShow: 0 }, description: false, item: false, footer: false } // 第一页总是显示header if (pageIndex === 1 && props.sectionStates.header) { pageContent.header = true currentHeight += 48 } // 计算table应该显示多少行 if (props.sectionStates.table) { const availableHeight = maxHeightPerPage - currentHeight - 100 const maxRowsForThisPage = Math.max(0, Math.floor((availableHeight - 40) / 15)) if (maxRowsForThisPage > 0) { pageContent.table.show = true pageContent.table.rowsToShow = Math.min(maxRowsForThisPage, totalRows) } } return pageContent }

2. 表格行分页

// 获取当前页面应该显示的表格行数 const getTableRowsForPage = (table: any, pageIndex: number) => { const pageContent = getPageContent(pageIndex) if (!pageContent.table.show) return [] const allRows = getTableRows(table) const rowsToShow = pageContent.table.rowsToShow // 计算从哪一行开始显示 let startIndex = 0 for (let i = 1; i < pageIndex; i++) { const prevPageContent = getPageContent(i) if (prevPageContent.table.show) { startIndex += prevPageContent.table.rowsToShow } } return allRows.slice(startIndex, startIndex + rowsToShow) }

3. 精确分页计算

// 重新设计分页计算逻辑 const totalPages = computed(() => { const maxHeightPerPage = 700 let totalHeight = 0 // 计算各部分高度 if (props.sectionStates.header) totalHeight += 48 if (props.sectionStates.info) totalHeight += items.length * 20 if (props.sectionStates.table) totalHeight += 40 + (totalRows * 15) // ... 其他部分 const pages = Math.ceil(totalHeight / maxHeightPerPage) return Math.max(1, pages) })

主要改进：
